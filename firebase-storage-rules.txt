# Firebase Storage Rules for Fashion AI App
# Copy these rules to your Firebase Console > Storage > Rules

rules_version = '2';

// Craft rules based on our app's requirements
service firebase.storage {
  match /b/{bucket}/o {
    // Helper functions
    function isAuthenticated() {
      return request.auth != null;
    }
    
    function isOwner(userId) {
      return request.auth.uid == userId;
    }
    
    function isValidImage() {
      return request.resource.contentType.matches('image/.*')
        && request.resource.size < 5 * 1024 * 1024; // 5MB max
    }
    
    function isValidPath(userId) {
      return request.resource.name.matches(userId + '/.*');
    }

    // User profile images
    match /users/{userId}/profile/{imageId} {
      allow read: if isAuthenticated();
      allow write: if isAuthenticated() 
        && isOwner(userId)
        && isValidImage();
    }
    
    // Wardrobe items
    match /wardrobe/{userId}/{itemId} {
      allow read: if isAuthenticated() && isOwner(userId);
      allow write: if isAuthenticated() 
        && isOwner(userId)
        && isValidImage()
        && isValidPath(userId);
    }
    
    // Style moments (public posts)
    match /moments/{momentId} {
      allow read: if isAuthenticated();
      allow create: if isAuthenticated() 
        && isValidImage()
        && isValidPath(request.auth.uid);
      allow delete: if isAuthenticated() 
        && isOwner(resource.metadata.userId);
    }
    
    // Temporary uploads (for processing)
    match /temp/{userId}/{fileName} {
      allow read, write: if isAuthenticated() 
        && isOwner(userId)
        && isValidImage();
        
      // Auto-delete after 24 hours
      match /{allPaths=**} {
        allow read, write: if isAuthenticated() 
          && isOwner(userId)
          && resource.timeCreated + duration.value(24, 'h') > request.time;
      }
    }
  }
}
